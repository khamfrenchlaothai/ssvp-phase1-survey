⚙️ Admin: Google Sheets backup setup (one-time, ~5 min)

This silently saves every submission to a Google Sheet — guaranteed backup even if WhatsApp/Email fails.

  1. Open sheets.new to create a fresh sheet.
  2. Menu: Extensions → Apps Script.
  3. Delete any existing code, paste the script below, click 💾 Save.
  4. Click Deploy → New deployment → ⚙️ → Web app.
  5. Settings: Execute as: Me · Who has access: Anyone. Click Deploy and authorize.
  6. Copy the Web App URL (looks like https://script.google.com/macros/s/.../exec).
  7. In index.html find const SHEETS_WEBAPP_URL = ""; and paste the URL between the quotes.
  8. Commit & push to GitHub. Done — every submission auto-backups.
Apps Script code to paste: 
function doPost(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  try {
    const d = JSON.parse(e.postData.contents);
    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        'Received','Survey ID','Survey Date','Interviewer','Beneficiary',
        'Phone','Village','GPS','HH Size','Children<5','Elderly 65+',
        'Family Professions',
        'Roof Damage','Electrical Risk','Damaged Area (m²)','Damage Score','Vulnerabilities',
        'Ownership',
        'Helpers Available','Helpers Count','Helpers Date',
        'Aluminum OK','Local Workers','Labor Cost (LAK/m²)',
        'Notes','Consent'
      ]);
    }
    sheet.appendRow([
      new Date(), d.surveyId, d.date, d.interviewer, d.beneficiary,
      d.phone, d.village, d.gps, d.hhSize, d.hhChildren, d.hhElderly,
      (d.memberJobs || []).join(' · '),
      d.roofText, d.elecText, d.damagedArea, d.score, d.vuln,
      d.ownership,
      d.helpersAvailable, d.helpersCount, d.helpersDate,
      d.aluminumOk, d.localWorkers, d.laborCost,
      d.notes, d.consent ? 'Yes' : 'No'
    ]);
    return ContentService.createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,err:String(err)}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
